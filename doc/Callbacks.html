def end
    @nodes.flush()

    @relations.flush()

def node(reader)
    attributes = {MID => reader.attribute(ID), LON => reader.attribute(LON), LAT => reader.attribute(LAT)}
    unless reader.empty_element?
        extract_children(reader, attributes)
    end
    #puts attributes.to_json
    @nodes.add(attributes)
end
